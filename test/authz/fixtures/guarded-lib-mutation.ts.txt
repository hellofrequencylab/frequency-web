import { createAdminClient } from '@/lib/supabase/admin'
import { requireAdmin } from '@/lib/auth/guards'

/** Deletes any note — but only after proving the caller is an admin. */
export async function deleteNoteAsAdmin(noteId: string): Promise<void> {
  await requireAdmin()
  const admin = createAdminClient()
  await admin.from('notes').delete()
  void noteId
}
